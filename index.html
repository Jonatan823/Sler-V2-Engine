function generarSnapshotSLER() { // Capturamos el contenedor principal de datos const areaCaptura = document.getElementById('data-stream') || document.querySelector('.main-content'); console.log("Iniciando aplanado de formato estable..."); html2canvas(areaCaptura, { backgroundColor: "#0a0a0c", // Mantiene tu estética oscura scale: 2, // Calidad optimizada para lectura logging: false, useCORS: true // Por si usas recursos externos }).then(canvas => { const link = document.createElement('a'); link.download = `SLER_V2_OUTPUT_${new Date().getTime()}.png`; link.href = canvas.toDataURL("image/png"); document.body.appendChild(link); link.click(); document.body.removeChild(link); console.log("Archivo exportado exitosamente."); }).catch(err => { console.error("Error en el motor de exportación:", err); alert("Error al generar el formato estable. Revisa la consola (F12)."); }); }
